// ── SPARKLES ──
function spark(x,y){
  ['✨','💖','🌸','⭐','💕','💫'].forEach((e,i)=>{
    const s=document.createElement('div');
    s.className='sparkle';s.textContent=e;
    s.style.cssText=`left:${x}px;top:${y}px`;
    const a=(Math.PI*2*i)/6,d=45+Math.random()*30;
    s.style.setProperty('--dx',Math.cos(a)*d+'px');
    s.style.setProperty('--dy',Math.sin(a)*d+'px');
    document.body.appendChild(s);
    setTimeout(()=>s.remove(),850);
  });
}
document.addEventListener('click',e=>spark(e.clientX,e.clientY));

// ── COUNTDOWN ──
function updateCD(){
  const now=new Date();
  let bd=new Date(now.getFullYear(),3,29);
  if(now>bd) bd=new Date(now.getFullYear()+1,3,29);
  const diff=bd-now;
  const d=Math.floor(diff/86400000),h=Math.floor((diff%86400000)/3600000),m=Math.floor((diff%3600000)/60000);
  const el=document.getElementById('countdown-text');

const stackEl=document.getElementById('cardStack');

let topIdx=0;

function buildStack(){
  stackEl.innerHTML='';cardEls=[];topIdx=0;
  const total=cardData.length;
  // Render last→first so card 0 is appended last and sits on top in DOM
  for(let i=total-1;i>=0;i--){
    const d=cardData[i];
    const card=document.createElement('div');
    card.className='swipe-card';
    // Card 0 = top of stack (offset 0, no rotation, scale 1, highest z-index)
    const off=Math.min(i,3);
    const rot=(i%2===0?-1:1)*off*2.2;
    // z-index: card 0 gets highest (total), card N-1 gets 1
    card.style.cssText=`transform:rotate(${rot}deg) translateY(${off*5}px) scale(${1-off*.035});z-index:${total-i};background:white;`;
    card.innerHTML=`
      <div class="card-img" style="background:${d.bg}">
        <div class="card-emoji" style="animation-delay:${i*.25}s">${d.emoji}</div>
      </div>
      <div class="card-body">
        <div class="card-tag" style="color:${d.tagC}">${d.tag}</div>
        <div class="card-title">${d.title}</div>
        <div class="card-desc">${d.desc}</div>
      </div>
      <div class="badge badge-love">LOVE 💖</div>
      <div class="badge badge-next">NEXT →</div>`;
    attachSwipe(card,i);
    stackEl.appendChild(card);
    cardEls[i]=card;
  }
  updateCounter();
}

function updateCounter(){
  const left=cardData.length-topIdx;
  if(left>0){
    counterEl.textContent=`${left} card${left>1?'s':''} remaining · swipe to reveal 💕`;
  } else {
    counterEl.innerHTML='✨ You\'ve seen all the love! <button onclick="buildStack()" style="margin-left:.5rem;padding:.4rem 1rem;border:2px solid var(--rose);border-radius:100px;background:white;color:var(--rose);font-family:\'DM Mono\',monospace;font-size:.6rem;letter-spacing:.08em;cursor:pointer">↺ Again</button>';
  }
}

function attachSwipe(card,idx){
  let sx,sy,isDrag=false,curX=0;

  function getP(e){return e.touches?{x:e.touches[0].clientX,y:e.touches[0].clientY}:{x:e.clientX,y:e.clientY};}

  function onStart(e){
    if(idx!==topIdx)return;
    const p=getP(e);sx=p.x;sy=p.y;isDrag=true;
    card.style.transition='none';
    e.preventDefault();
  }
  function onMove(e){
    if(!isDrag||idx!==topIdx)return;
    const p=getP(e);
    curX=p.x-sx;
    const dy=(p.y-sy)*.25;
    const rot=curX*.07;
    card.style.transform=`translateX(${curX}px) translateY(${dy}px) rotate(${rot}deg)`;
    const pct=Math.min(Math.abs(curX)/(window.innerWidth*.3),1);
    card.querySelector('.badge-love').style.opacity=curX>0?pct:0;
    card.querySelector('.badge-next').style.opacity=curX<0?pct:0;
    e.preventDefault();
  }
  function onEnd(){
    if(!isDrag||idx!==topIdx)return;
    isDrag=false;
    card.style.transition='transform .38s cubic-bezier(.25,.46,.45,.94), opacity .38s ease';
    const threshold=window.innerWidth*.25;
    if(Math.abs(curX)>threshold){
      const dir=curX>0?1:-1;
      card.style.transform=`translateX(${dir*window.innerWidth*1.5}px) rotate(${dir*28}deg)`;
      card.style.opacity='0';
      spark(window.innerWidth*(dir>0?.75:.25),window.innerHeight*.4);
      setTimeout(()=>card.style.display='none',400);
      topIdx++;
      updateCounter();
      for(let k=topIdx;k<Math.min(topIdx+4,cardEls.length);k++){
        const newOff=k-topIdx;
        const newRot=(k%2===0?-1:1)*Math.min(newOff,3)*2.2;
        const newY=Math.min(newOff,3)*5;
        const newScale=1-Math.min(newOff,3)*.035;
        setTimeout(()=>{
          cardEls[k].style.transition='transform .35s ease';
          cardEls[k].style.transform=`rotate(${newRot}deg) translateY(${newY}px) scale(${newScale})`;
        },50);
      }
    } else {
      const off=Math.min(idx,3);
      const rot=(idx%2===0?-1:1)*off*2.2;
      card.style.transform=`rotate(${rot}deg) translateY(${off*5}px) scale(${1-off*.035})`;
      card.querySelector('.badge-love').style.opacity=0;
      card.querySelector('.badge-next').style.opacity=0;
    }
    curX=0;
  }

  card.addEventListener('mousedown',onStart);
  card.addEventListener('touchstart',onStart,{passive:false});
  document.addEventListener('mousemove',onMove);
  document.addEventListener('touchmove',onMove,{passive:false});
  document.addEventListener('mouseup',onEnd);
  document.addEventListener('touchend',onEnd);
}

let fwP=[],fwOn=false;
class FP{
  constructor(x,y,c){
    this.x=x;this.y=y;
    const a=Math.random()*Math.PI*2,s=2+Math.random()*5;
    this.vx=Math.cos(a)*s;this.vy=Math.sin(a)*s;
    this.a=1;this.c=c;this.r=2+Math.random()*3;this.g=.07;
  }
  u(){this.x+=this.vx;this.y+=this.vy;this.vy+=this.g;this.a-=.013;this.vx*=.98;}
  d(){fwX.save();fwX.globalAlpha=this.a;fwX.fillStyle=this.c;fwX.beginPath();fwX.arc(this.x,this.y,this.r,0,Math.PI*2);fwX.fill();fwX.restore();}
}
function explode(x,y){['#e8556d','#f9c4d0','#d4a847','#c9b8e8','#fff'].forEach(c=>{for(let i=0;i<15;i++)fwP.push(new FP(x,y,c));});}
function animFW(){
  if(!fwOn)return;
  fwX.clearRect(0,0,fwC.width,fwC.height);
  fwP=fwP.filter(p=>p.a>0);fwP.forEach(p=>{p.u();p.d();});
  if(fwP.length>0)requestAnimationFrame(animFW);else fwOn=false;
}
function launchFW(){fwOn=true;animFW();let n=0;const t=setInterval(()=>{explode(Math.random()*fwC.width,Math.random()*fwC.height*.6);if(++n>20)clearInterval(t);},280);}
setTimeout(()=>{fwOn=true;animFW();[0,320,640,960,1280].forEach((dl,i)=>setTimeout(()=>explode(fwC.width*(.2+Math.random()*.6),fwC.height*(.1+Math.random()*.4)),dl));},1100);
